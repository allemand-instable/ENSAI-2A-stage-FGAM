% & Name
% 		\info
% & Description
% 		Displays an environment delimited with a blue line on the left, with an Info Icon located at the left of the line
% & Arguments
% 		1: text
% & Examples
% 		\info{Hello World}
%       \info{\smallskip\centering
%           Text with some mathematics : $\int f \, d\mu$
%       }
\newcommand{\info}[1]{%
    \awesomebox[flatuicolors_blue]{2pt}{\faInfoCircle}{flatuicolors_blue}{#1}%
}
